"use client";

import Link from "next/link";
import {
  Bell,
  Building2,
  RefreshCw,
  Settings,
  Shield,
  SlidersHorizontal,
  Users,
} from "lucide-react";
import { useEffect, useMemo, useState } from "react";

import ProtecaoModulo from "@/components/ProtecaoModulo";
import SigActionCard from "@/components/sig/SigActionCard";
import SigButton from "@/components/sig/SigButton";
import SigCard from "@/components/sig/SigCard";
import SigPageHeader from "@/components/sig/SigPageHeader";
import SigStatCard from "@/components/sig/SigStatCard";

import {
  lerMunicipioContextoLocal,
  obterMunicipioIdEfetivo,
} from "@/lib/contextoMunicipio";
import { supabase } from "@/lib/supabase";

type UsuarioLocal = {
  perfil?: string;
  municipio_id?: number;
};

type TotaisConfiguracoes = {
  municipios: number;
  usuarios: number;
  notificacoes: number;
  notificacoesNaoLidas: number;
};

const itens = [
  {
    titulo: "Configurações",
    href: "/sistema/configuracoes",
    descricao:
      "Parâmetros gerais, identidade institucional e preferências do município.",
    icone: Settings,
    detalhe: "Abrir configurações",
  },
  {
    titulo: "Municípios",
    href: "/sistema/municipios",
    descricao:
      "Gerenciamento dos municípios cadastrados e suas informações oficiais.",
    icone: Building2,
    detalhe: "Gerenciar municípios",
  },
  {
    titulo: "Permissões",
    href: "/sistema/usuarios/permissoes",
    descricao:
      "Controle de acesso por perfil, módulo e município.",
    icone: Shield,
    detalhe: "Gerenciar permissões",
  },
  {
    titulo: "Notificações",
    href: "/sistema/notificacoes",
    descricao:
      "Gerenciamento de alertas, avisos e notificações do sistema.",
    icone: Bell,
    detalhe: "Abrir notificações",
  },
];

function obterUsuarioLocal(): UsuarioLocal | null {
  if (typeof window === "undefined") return null;

  try {
    return JSON.parse(
      localStorage.getItem("usuarioLogado") || "null"
    ) as UsuarioLocal | null;
  } catch {
    return null;
  }
}

export default function ConfiguracoesCentralPage() {
  const [totais, setTotais] = useState<TotaisConfiguracoes>({
    municipios: 0,
    usuarios: 0,
    notificacoes: 0,
    notificacoesNaoLidas: 0,
  });

  const [municipioNome, setMunicipioNome] =
    useState("Município");
  const [perfil, setPerfil] = useState("");
  const [carregando, setCarregando] = useState(true);
  const [erro, setErro] = useState("");

  async function carregar() {
    setCarregando(true);
    setErro("");

    try {
      const usuario = obterUsuarioLocal();

      if (!usuario?.perfil) {
        throw new Error("Usuário não identificado.");
      }

      const contexto = lerMunicipioContextoLocal();

      const municipioId = obterMunicipioIdEfetivo({
        perfil: usuario.perfil,
        municipioIdUsuario: usuario.municipio_id,
      });

      if (!municipioId) {
        throw new Error("Município não identificado.");
      }

      const [
        municipioResposta,
        municipiosResposta,
        usuariosResposta,
        notificacoesResposta,
        notificacoesNaoLidasResposta,
      ] = await Promise.all([
        supabase
          .from("municipios")
          .select("nome")
          .eq("id", municipioId)
          .maybeSingle(),

        supabase
          .from("municipios")
          .select("id", {
            count: "exact",
            head: true,
          }),

        supabase
          .from("usuarios")
          .select("id", {
            count: "exact",
            head: true,
          })
          .eq("municipio_id", municipioId),

        supabase
          .from("notificacoes")
          .select("id", {
            count: "exact",
            head: true,
          })
          .eq("municipio_id", municipioId),

        supabase
          .from("notificacoes")
          .select("id", {
            count: "exact",
            head: true,
          })
          .eq("municipio_id", municipioId)
          .eq("lida", false),
      ]);

      if (municipioResposta.error) {
        console.warn(
          "Falha parcial em municipios:",
          municipioResposta.error.message
        );
      }

      const falhas = [
        municipiosResposta.error,
        usuariosResposta.error,
        notificacoesResposta.error,
        notificacoesNaoLidasResposta.error,
      ].filter(Boolean);

      for (const falha of falhas) {
        console.warn(
          "Falha parcial na Central de Configurações:",
          falha?.message
        );
      }

      setPerfil(
        String(usuario.perfil || "").toUpperCase()
      );

      setMunicipioNome(
        String(
          municipioResposta.data?.nome ||
            contexto?.nome ||
            "Município"
        )
      );

      setTotais({
        municipios: municipiosResposta.count || 0,
        usuarios: usuariosResposta.count || 0,
        notificacoes: notificacoesResposta.count || 0,
        notificacoesNaoLidas:
          notificacoesNaoLidasResposta.count || 0,
      });
    } catch (error) {
      console.error(
        "Erro ao carregar Central de Configurações:",
        error
      );

      setErro(
        error instanceof Error
          ? error.message
          : "Não foi possível carregar a Central de Configurações."
      );
    } finally {
      setCarregando(false);
    }
  }

  useEffect(() => {
    void carregar();
  }, []);

  const podeGerenciarMunicipios = useMemo(
    () => perfil === "DESENVOLVEDOR",
    [perfil]
  );

  const itensFiltrados = useMemo(
    () =>
      itens.filter(
        (item) =>
          item.href !== "/sistema/municipios" ||
          podeGerenciarMunicipios
      ),
    [podeGerenciarMunicipios]
  );

  return (
    <ProtecaoModulo modulo="configuracoes">
      <main className="sig-page">
        <div className="sig-page-content">
          <SigPageHeader
            titulo="Central de Configurações"
            subtitulo={`${municipioNome} • Parâmetros gerais, permissões, identidade institucional e notificações.`}
            detalhe="Administração do sistema"
            icone={SlidersHorizontal}
            acoes={
              <>
                <Link href="/sistema/configuracoes">
                  <SigButton
                    type="primary"
                    icon={Settings}
                    size="sm"
                  >
                    Configurações
                  </SigButton>
                </Link>

                <SigButton
                  type="cyan"
                  icon={RefreshCw}
                  size="sm"
                  loading={carregando}
                  onClick={() => void carregar()}
                >
                  Atualizar
                </SigButton>
              </>
            }
          />

          {erro ? <div className="sig-error">{erro}</div> : null}

          <section className="grid gap-4 sm:grid-cols-2 xl:grid-cols-5">
            <SigStatCard
              titulo="Município atual"
              valor={municipioNome}
              subtitulo="Contexto ativo"
              icone={Building2}
              destaque="cyan"
            />

            <SigStatCard
              titulo="Usuários"
              valor={totais.usuarios}
              subtitulo="Vinculados ao município"
              icone={Users}
              destaque="green"
            />

            <SigStatCard
              titulo="Notificações"
              valor={totais.notificacoes}
              subtitulo="Registros do município"
              icone={Bell}
              destaque="blue"
            />

            <SigStatCard
              titulo="Não lidas"
              valor={totais.notificacoesNaoLidas}
              subtitulo="Dependem de atenção"
              icone={Bell}
              destaque="red"
            />

            <SigStatCard
              titulo="Perfil atual"
              valor={perfil || "-"}
              subtitulo="Nível de configuração"
              icone={Shield}
              destaque="slate"
            />
          </section>

          {carregando ? (
            <div className="sig-loading">
              <div>
                <div className="mx-auto h-10 w-10 animate-spin rounded-full border-4 border-slate-700 border-t-cyan-400" />

                <p className="mt-4 text-slate-400">
                  Carregando configurações...
                </p>
              </div>
            </div>
          ) : (
            <>
              <section className="grid gap-4 xl:grid-cols-12">
                <SigCard className="xl:col-span-7">
                  <CabecalhoSecao
                    titulo="Configurações institucionais"
                    subtitulo="Principais áreas de parametrização"
                    icone={Settings}
                  />

                  <div className="mt-5 grid gap-3 md:grid-cols-2">
                    <Painel
                      titulo="Identidade do município"
                      descricao="Nome da Guarda, brasões, comandante, cores e informações institucionais."
                    />

                    <Painel
                      titulo="Acesso e permissões"
                      descricao="Perfis, módulos liberados e regras de segurança."
                    />

                    <Painel
                      titulo="Notificações"
                      descricao="Alertas internos, avisos e preferências de comunicação."
                    />

                    <Painel
                      titulo="Parâmetros locais"
                      descricao="Configurações específicas da operação municipal."
                    />
                  </div>
                </SigCard>

                <SigCard className="xl:col-span-5" destaque>
                  <CabecalhoSecao
                    titulo="Segurança"
                    subtitulo="Regras essenciais da configuração"
                    icone={Shield}
                  />

                  <div className="mt-5 space-y-3">
                    <Regra texto="Configurações devem respeitar o município ativo." />
                    <Regra texto="Permissões devem ser aplicadas por perfil e módulo." />
                    <Regra texto="Alterações sensíveis devem ser auditadas." />
                    <Regra texto="O perfil DESENVOLVEDOR mantém acesso global." />
                  </div>
                </SigCard>
              </section>

              <section>
                <div className="mb-4">
                  <h2 className="text-2xl font-black text-white">
                    Ferramentas de Configuração
                  </h2>

                  <p className="mt-1 text-sm text-slate-400">
                    Acesse os recursos disponíveis para o seu perfil.
                  </p>
                </div>

                <div className="grid gap-4 md:grid-cols-2 xl:grid-cols-4">
                  {itensFiltrados.map((item) => (
                    <SigActionCard
                      key={item.href}
                      titulo={item.titulo}
                      descricao={item.descricao}
                      href={item.href}
                      icone={item.icone}
                      detalhe={item.detalhe}
                    />
                  ))}
                </div>
              </section>
            </>
          )}
        </div>
      </main>
    </ProtecaoModulo>
  );
}

function CabecalhoSecao({
  titulo,
  subtitulo,
  icone: Icone,
}: {
  titulo: string;
  subtitulo: string;
  icone: typeof Settings;
}) {
  return (
    <div className="flex items-center gap-3 border-b border-slate-800 pb-4">
      <div className="flex h-11 w-11 shrink-0 items-center justify-center rounded-xl border border-cyan-400/20 bg-cyan-400/[0.07] text-cyan-300">
        <Icone className="h-5 w-5" />
      </div>

      <div>
        <h2 className="font-black text-white">{titulo}</h2>
        <p className="mt-0.5 text-xs text-slate-500">
          {subtitulo}
        </p>
      </div>
    </div>
  );
}

function Painel({
  titulo,
  descricao,
}: {
  titulo: string;
  descricao: string;
}) {
  return (
    <div className="rounded-2xl border border-slate-800 bg-slate-950/45 p-4">
      <h3 className="font-black text-white">{titulo}</h3>

      <p className="mt-2 text-sm leading-6 text-slate-400">
        {descricao}
      </p>
    </div>
  );
}

function Regra({ texto }: { texto: string }) {
  return (
    <div className="flex items-start gap-3 rounded-xl border border-cyan-400/15 bg-cyan-400/[0.04] p-3">
      <Shield className="mt-0.5 h-4 w-4 shrink-0 text-cyan-300" />
      <span className="text-sm text-slate-300">{texto}</span>
    </div>
  );
}
